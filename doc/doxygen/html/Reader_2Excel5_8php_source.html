00967

02136       $hasRichText = (($optionFlags & 0x08) != 0);

02139         // number of Rich-Text formatting runs

02142       }
02143 
02144       if ($hasAsian) {

02149 
02150       // expected byte length of character array if not split
02151       $len = ($isCompressed) ? $numChars : $numChars * 2;
02152 
02153       // look up limit position
02154       foreach ($spliceOffsets as $spliceOffset) {
02155         // it can happen that the string is empty, therefore we need
02156         // <= and not just <
02157         if ($pos <= $spliceOffset) {
02158           $limitpos = $spliceOffset;
02159           break;

02161       }
02162 
02163       if ($pos + $len <= $limitpos) {
02164         // character array is not split between records

02168

02181

02183         while ($charsLeft > 0) {
02184

02186           foreach ($spliceOffsets as $spliceOffset) {
02187             if ($pos < $spliceOffset) {

02189               break;
02190             }

02210             $retstr .= substr($recordData, $pos, $len);
02211             $charsLeft -= $len / 2;
02212             $isCompressed = false;
02213 
02214           } elseif (!$isCompressed && ($option == 0)) {
02215             // 1st fragment uncompressed
02216             // this fragment compressed
02217             $len = min($charsLeft, $limitpos - $pos);
02218             for ($j = 0; $j < $len; ++$j) {
02219               $retstr .= $recordData{$pos + $j} . chr(0);
02220             }
02221             $charsLeft -= $len;
02222             $isCompressed = false;
02223 
02224           } else {
02225             // 1st fragment compressed
02226             // this fragment uncompressed
02227             $newstr = '';
02228             for ($j = 0; $j < strlen($retstr); ++$j) {
02229               $newstr .= $retstr[$j] . chr(0);
02230             }
02231             $retstr = $newstr;
02232             $len = min($charsLeft * 2, $limitpos - $pos);
02233             $retstr .= substr($recordData, $pos, $len);
02234             $charsLeft -= $len / 2;
02235             $isCompressed = false;
02236           }
02237 
02238           $pos += $len;
02239         }

02242       // convert to UTF-8
02243       $retstr = $this->_encodeUTF16($retstr, $isCompressed);
02244

02246       $fmtRuns = array();

03062   {

03064     $recordData = substr($this->_data, $this->_pos + 4, $length);
03065 
03066     // move stream pointer to next record

03068

03071

03073     $column = $this->_GetInt2d($recordData, 2);

03075 
03076     // offset: 20: size: variable; formula structure
03077     $formulaStructure = substr($recordData, 20);
03078 
03079     // offset: 14: size: 2; option flags, recalculate always, recalculate on open etc.
03080     $options = $this->_GetInt2d($recordData, 14);
03081 
03082     // bit: 0; mask: 0x0001; 1 = recalculate always
03083     // bit: 1; mask: 0x0002; 1 = calculate on open

03086 
03087     // WARNING:
03088     // We can apparently not rely on $isPartOfSharedFormula. Even when $isPartOfSharedFormula = true
03089     // the formula data may be ordinary formula data, therefore we need to check
03090     // explicitly for the tExp token (0x01)
03091     $isPartOfSharedFormula = $isPartOfSharedFormula && ord($formulaStructure{2}) == 0x01;
03092 
03093     if ($isPartOfSharedFormula) {
03094       // part of shared formula which means there will be a formula with a tExp token and nothing else
03095       // get the base cell, grab tExp token
03096       $baseRow = $this->_GetInt2d($formulaStructure, 3);
03097       $baseCol = $this->_GetInt2d($formulaStructure, 5);
03098       $this->_baseCell = PHPExcel_Cell::stringFromColumnIndex($baseCol). ($baseRow + 1);

03103

03105         // formula is added to this cell after the sheet has been read
03106         $this->_sharedFormulaParts[$columnString . ($row + 1)] = $this->_baseCell;
03107       }
03108 
03109       // offset: 16: size: 4; not used
03110 
03111       // offset: 4; size: 2; XF index
03112       $xfIndex = $this->_GetInt2d($recordData, 4);
03113 
03114       // offset: 6; size: 8; result of the formula

03118

03121

03140         && (ord($recordData{12}) == 255)
03141         && (ord($recordData{13}) == 255)) {
03142

03144         $dataType = PHPExcel_Cell_DataType::TYPE_ERROR;

03146

03148         && (ord($recordData{12}) == 255)
03149         && (ord($recordData{13}) == 255)) {
03150 
03151         // Formula result is a null string
03152         $dataType = PHPExcel_Cell_DataType::TYPE_NULL;

03154 
03155       } else {
03156

03167       }
03168

03170       if (!$isPartOfSharedFormula) {
03171         // not part of shared formula
03172         // add cell value. If we can read formula, populate with formula, otherwise just used cached value

03175             throw new Exception('Not BIFF8. Can only read BIFF8 formulas');

03183       } else {

03185           // do nothing at this point, formula id added later in the code

03187           $cell->setValueExplicit($value, $dataType);
03188         }
03189       }
03190

03192       $cell->setCalculatedValue($value);

03632   private function _readHyperLink()
03633   {
03634     $length = $this->_GetInt2d($this->_data, $this->_pos + 2);
03635     $recordData = substr($this->_data, $this->_pos + 4, $length);
03636 
03637     // move stream pointer forward to next record
03638     $this->_pos += 4 + $length;
03639 
03640     if (!$this->_readDataOnly) {
03641       // offset: 0; size: 8; cell range address of all cells containing this hyperlink
03642       try {
03643         $cellRange = $this->_readBIFF8CellRangeAddressFixed($recordData, 0, 8);
03644       } catch (Exception $e) {
03645         return;
03646       }
03647 
03648       // offset: 8, size: 16; GUID of StdLink
03649 
03650       // offset: 24, size: 4; unknown value
03651 
03652       // offset: 28, size: 4; option flags
03653 
03654         // bit: 0; mask: 0x00000001; 0 = no link or extant, 1 = file link or URL
03655         $isFileLinkOrUrl = (0x00000001 & $this->_GetInt2d($recordData, 28)) >> 0;
03656 
03657         // bit: 1; mask: 0x00000002; 0 = relative path, 1 = absolute path or URL
03658         $isAbsPathOrUrl = (0x00000001 & $this->_GetInt2d($recordData, 28)) >> 1;
03659 
03660         // bit: 2 (and 4); mask: 0x00000014; 0 = no description
03661         $hasDesc = (0x00000014 & $this->_GetInt2d($recordData, 28)) >> 2;
03662 
03663         // bit: 3; mask: 0x00000008; 0 = no text, 1 = has text
03664         $hasText = (0x00000008 & $this->_GetInt2d($recordData, 28)) >> 3;
03665 
03666         // bit: 7; mask: 0x00000080; 0 = no target frame, 1 = has target frame
03667         $hasFrame = (0x00000080 & $this->_GetInt2d($recordData, 28)) >> 7;
03668 
03669         // bit: 8; mask: 0x00000100; 0 = file link or URL, 1 = UNC path (inc. server name)
03670         $isUNC = (0x00000100 & $this->_GetInt2d($recordData, 28)) >> 8;
03671 
03672       // offset within record data
03673       $offset = 32;
03674 
03675       if ($hasDesc) {
03676         // offset: 32; size: var; character count of description text
03677         $dl = $this->_GetInt4d($recordData, 32);
03678         // offset: 36; size: var; character array of description text, no Unicode string header, always 16-bit characters, zero terminated
03679         $desc = $this->_encodeUTF16(substr($recordData, 36, 2 * ($dl - 1)), false);
03680         $offset += 4 + 2 * $dl;
03681       }
03682       if ($hasFrame) {
03683         $fl = $this->_GetInt4d($recordData, $offset);
03684         $offset += 4 + 2 * $fl;
03685       }
03686 
03687       // detect type of hyperlink (there are 4 types)
03688       $hyperlinkType = null;
03689 
03690       if ($isUNC) {
03691         $hyperlinkType = 'UNC';
03692       } else if (!$isFileLinkOrUrl) {
03693         $hyperlinkType = 'workbook';
03694       } else if (ord($recordData{$offset}) == 0x03) {
03695         $hyperlinkType = 'local';
03696       } else if (ord($recordData{$offset}) == 0xE0) {
03697         $hyperlinkType = 'URL';
03698       }
03699 
03700       switch ($hyperlinkType) {
03701       case 'URL':
03702         // section 5.58.2: Hyperlink containing a URL
03703         // e.g. http://example.org/index.php
03704 
03705         // offset: var; size: 16; GUID of URL Moniker
03706         $offset += 16;
03707         // offset: var; size: 4; size (in bytes) of character array of the URL including trailing zero word
03708         $us = $this->_GetInt4d($recordData, $offset);
03709         $offset += 4;
03710         // offset: var; size: $us; character array of the URL, no Unicode string header, always 16-bit characters, zero-terminated
03711         $url = $this->_encodeUTF16(substr($recordData, $offset, $us - 2), false);
03712         $url .= $hasText ? '#' : '';
03713         $offset += $us;
03714         break;
03715 
03716       case 'local':
03717         // section 5.58.3: Hyperlink to local file
03718         // examples:
03719         //   mydoc.txt
03720         //   ../../somedoc.xls#Sheet!A1
03721 
03722         // offset: var; size: 16; GUI of File Moniker
03723         $offset += 16;
03724 
03725         // offset: var; size: 2; directory up-level count.
03726         $upLevelCount = $this->_GetInt2d($recordData, $offset);
03727         $offset += 2;
03728 
03729         // offset: var; size: 4; character count of the shortened file path and name, including trailing zero word
03730         $sl = $this->_GetInt4d($recordData, $offset);
03731         $offset += 4;
03732 
03733         // offset: var; size: sl; character array of the shortened file path and name in 8.3-DOS-format (compressed Unicode string)
03734         $shortenedFilePath = substr($recordData, $offset, $sl);
03735         $shortenedFilePath = $this->_encodeUTF16($shortenedFilePath, true);
03736         $shortenedFilePath = substr($shortenedFilePath, 0, -1); // remove trailing zero
03737 
03738         $offset += $sl;
03739 
03740         // offset: var; size: 24; unknown sequence
03741         $offset += 24;
03742 
03743         // extended file path
03744         // offset: var; size: 4; size of the following file link field including string lenth mark
03745         $sz = $this->_GetInt4d($recordData, $offset);
03746         $offset += 4;
03747 
03748         // only present if $sz > 0
03749         if ($sz > 0) {
03750           // offset: var; size: 4; size of the character array of the extended file path and name
03751           $xl = $this->_GetInt4d($recordData, $offset);
03752           $offset += 4;
03753 
03754           // offset: var; size 2; unknown
03755           $offset += 2;
03756 
03757           // offset: var; size $xl; character array of the extended file path and name.
03758           $extendedFilePath = substr($recordData, $offset, $xl);
03759           $extendedFilePath = $this->_encodeUTF16($extendedFilePath, false);
03760           $offset += $xl;
03761         }
03762 
03763         // construct the path
03764         $url = str_repeat('..\\', $upLevelCount);
03765         $url .= ($sz > 0) ?
03766           $extendedFilePath : $shortenedFilePath; // use extended path if available
03767         $url .= $hasText ? '#' : '';
03768 
03769         break;
03770 
03771 
03772       case 'UNC':
03773         // section 5.58.4: Hyperlink to a File with UNC (Universal Naming Convention) Path
03774         // todo: implement
03775         return;
03776 
03777       case 'workbook':
03778         // section 5.58.5: Hyperlink to the Current Workbook
03779         // e.g. Sheet2!B1:C2, stored in text mark field
03780         $url = 'sheet://';
03781         break;
03782 
03783       default:
03784         return;
03785 
03786       }
03787 
03788       if ($hasText) {
03789         // offset: var; size: 4; character count of text mark including trailing zero word
03790         $tl = $this->_GetInt4d($recordData, $offset);
03791         $offset += 4;
03792         // offset: var; size: var; character array of the text mark without the # sign, no Unicode header, always 16-bit characters, zero-terminated
03793         $text = $this->_encodeUTF16(substr($recordData, $offset, 2 * ($tl - 1)), false);
03794         $url .= $text;
03795       }
03796 
03797       // apply the hyperlink to all the relevant cells
03798       foreach (PHPExcel_Cell::extractAllCellReferencesInRange($cellRange) as $coordinate) {
03799         $this->_phpSheet->getCell($coordinate)->getHyperLink()->setUrl($url);
03800       }
03801     }
03802   }

04417   private function _createFormulaFromTokens($tokens, $additionalData)
04418   {

04421       return '';
04422     }

04426       // initialize spaces
04427       $space0 = isset($space0) ? $space0 : ''; // spaces before next token, not tParen
04428       $space1 = isset($space1) ? $space1 : ''; // carriage returns before next token, not tParen
04429       $space2 = isset($space2) ? $space2 : ''; // spaces before opening parenthesis
04430       $space3 = isset($space3) ? $space3 : ''; // carriage returns before opening parenthesis
04431       $space4 = isset($space4) ? $space4 : ''; // spaces before closing parenthesis
04432       $space5 = isset($space5) ? $space5 : ''; // carriage returns before closing parenthesis
04433 
04434       switch ($token['name']) {
04435       case 'tAdd': // addition
04436       case 'tConcat': // addition
04437       case 'tDiv': // division
04438       case 'tEQ': // equaltiy
04439       case 'tGE': // greater than or equal
04440       case 'tGT': // greater than
04441       case 'tIsect': // intersection
04442       case 'tLE': // less than or equal
04443       case 'tList': // less than or equal
04444       case 'tLT': // less than
04445       case 'tMul': // multiplication
04446       case 'tNE': // multiplication
04447       case 'tPower': // power
04448       case 'tRange': // range
04449       case 'tSub': // subtraction
04450         $op2 = array_pop($formulaStrings);
04451         $op1 = array_pop($formulaStrings);
04452         $formulaStrings[] = "$op1$space1$space0{$token['data']}$op2";
04453         unset($space0, $space1);
04454         break;
04455       case 'tUplus': // unary plus
04456       case 'tUminus': // unary minus
04457         $op = array_pop($formulaStrings);
04458         $formulaStrings[] = "$space1$space0{$token['data']}$op";
04459         unset($space0, $space1);
04460         break;
04461       case 'tPercent': // percent sign
04462         $op = array_pop($formulaStrings);
04463         $formulaStrings[] = "$op$space1$space0{$token['data']}";
04464         unset($space0, $space1);
04465         break;
04466       case 'tAttrVolatile': // indicates volatile function
04467       case 'tAttrIf':
04468       case 'tAttrSkip':
04469       case 'tAttrChoose':
04470         // token is only important for Excel formula evaluator
04471         // do nothing
04472         break;
04473       case 'tAttrSpace': // space / carriage return
04474         // space will be used when next token arrives, do not alter formulaString stack
04475         switch ($token['data']['spacetype']) {
04476         case 'type0':
04477           $space0 = str_repeat(' ', $token['data']['spacecount']);
04478           break;
04479         case 'type1':

04481           break;

04484           break;

04487           break;

04489           $space4 = str_repeat(' ', $token['data']['spacecount']);
04490           break;
04491         case 'type5':
04492           $space5 = str_repeat("\n", $token['data']['spacecount']);
04493           break;
04494         }
04495         break;
04496       case 'tAttrSum': // SUM function with one parameter
04497         $op = array_pop($formulaStrings);
04498         $formulaStrings[] = "{$space1}{$space0}SUM($op)";
04499         unset($space0, $space1);
04500         break;
04501       case 'tFunc': // function with fixed number of arguments
04502       case 'tFuncV': // function with variable number of arguments
04503         if ($token['data']['function'] != '') {
04504           // normal function
04505           $ops = array(); // array of operators
04506           for ($i = 0; $i < $token['data']['args']; ++$i) {
04507             $ops[] = array_pop($formulaStrings);
04508           }
04509           $ops = array_reverse($ops);
04510           $formulaStrings[] = "$space1$space0{$token['data']['function']}(" . implode(',', $ops) . ")";
04511           unset($space0, $space1);
04512         } else {

04517           }
04518           $ops = array_reverse($ops);
04519           $function = array_pop($formulaStrings);
04520           $formulaStrings[] = "$space1$space0$function(" . implode(',', $ops) . ")";
04521           unset($space0, $space1);
04522         }
04523         break;
04524       case 'tParen': // parenthesis
04525         $expression = array_pop($formulaStrings);
04526         $formulaStrings[] = "$space3$space2($expression$space5$space4)";

05806   private function _GetIEEE754($rknum)
05807   {
05808     if (($rknum & 0x02) != 0) {
05809       $value = $rknum >> 2;
05810     }
05811     else {
05812       // changes by mmp, info on IEEE754 encoding from
05813       // research.microsoft.com/~hollasch/cgindex/coding/ieeefloat.html
05814       // The RK format calls for using only the most significant 30 bits
05815       // of the 64 bit floating point value. The other 34 bits are assumed
05816       // to be 0 so we use the upper 30 bits of $rknum as follows...
05817       $sign = ($rknum & 0x80000000) >> 31;
05818       $exp = ($rknum & 0x7ff00000) >> 20;
05819       $mantissa = (0x100000 | ($rknum & 0x000ffffc));
05820       $value = $mantissa / pow( 2 , (20- ($exp - 1023)));
05821       if ($sign) {
05822         $value = -1 * $value;
05823       }
05824       //end of changes by mmp
05825     }
05826     if (($rknum & 0x01) != 0) {
05827       $value /= 100;
05828     }
05829     return $value;
