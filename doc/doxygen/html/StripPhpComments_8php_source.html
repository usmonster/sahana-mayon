00081         //            [^/*][^*]*\*+

00083         //                      ##    but do end with '*'
00084         //          /           ##  End of /* ... */ comment

00086         //        |         ##     OR  various things which aren't comments:
00087         //
00088         //          (

00090         //            (
00091         //              \\.           ##  Escaped char
00092         //            |               ##    OR
00093         //              [^"\\]        ##  Non "\
00094         //            )*
00095         //           "           ##  End of " ... " string
00096         //
00097         //          |         ##     OR
00098         //
00099         //            '           ##  Start of ' ... ' string
00100         //            (
00101         //              \\.           ##  Escaped char
00102         //            |               ##    OR
00103         //              [^'\\]        ##  Non '\
00104         //            )*
00105         //            '           ##  End of ' ... ' string

00107         //          |         ##     OR
