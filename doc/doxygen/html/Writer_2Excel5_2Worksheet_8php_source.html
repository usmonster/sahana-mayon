00837           $stringValue = $calculatedValue;
00838           $num = pack('CCCvCv', 0x00, 0x00, 0x00, 0x00, 0x00, 0xFFFF);
00839 
00840         }
00841 
00842       } else {

02606   function _processBitmap($bitmap)
02607   {
02608     // Open file.
02609     $bmp_fd = @fopen($bitmap,"rb");
02610     if (!$bmp_fd) {
02611       throw new Exception("Couldn't import $bitmap");
02612     }
02613 
02614     // Slurp the file into a string.
02615     $data = fread($bmp_fd, filesize($bitmap));
02616 
02617     // Check that the file is big enough to be a bitmap.
02618     if (strlen($data) <= 0x36) {
02619       throw new Exception("$bitmap doesn't contain enough data.\n");
02620     }
02621 
02622     // The first 2 bytes are used to identify the bitmap.
02623     $identity = unpack("A2ident", $data);
02624     if ($identity['ident'] != "BM") {
02625       throw new Exception("$bitmap doesn't appear to be a valid bitmap image.\n");
02626     }
02627 
02628     // Remove bitmap data: ID.
02629     $data = substr($data, 2);
02630 
02631     // Read and remove the bitmap size. This is more reliable than reading
02632     // the data size at offset 0x22.
02633     //
02634     $size_array   = unpack("Vsa", substr($data, 0, 4));
02635     $size   = $size_array['sa'];
02636     $data   = substr($data, 4);
02637     $size  -= 0x36; // Subtract size of bitmap header.
02638     $size  += 0x0C; // Add size of BIFF header.
02639 
02640     // Remove bitmap data: reserved, offset, header length.
02641     $data = substr($data, 12);
02642 
02643     // Read and remove the bitmap width and height. Verify the sizes.
02644     $width_and_height = unpack("V2", substr($data, 0, 8));
02645     $width  = $width_and_height[1];
02646     $height = $width_and_height[2];
02647     $data   = substr($data, 8);
02648     if ($width > 0xFFFF) {
02649       throw new Exception("$bitmap: largest image width supported is 65k.\n");
02650     }
02651     if ($height > 0xFFFF) {
02652       throw new Exception("$bitmap: largest image height supported is 65k.\n");
02653     }
02654 
02655     // Read and remove the bitmap planes and bpp data. Verify them.
02656     $planes_and_bitcount = unpack("v2", substr($data, 0, 4));
02657     $data = substr($data, 4);
02658     if ($planes_and_bitcount[2] != 24) { // Bitcount
02659       throw new Exception("$bitmap isn't a 24bit true color bitmap.\n");
02660     }
02661     if ($planes_and_bitcount[1] != 1) {
02662       throw new Exception("$bitmap: only 1 plane supported in bitmap image.\n");
02663     }
02664 
02665     // Read and remove the bitmap compression. Verify compression.
02666     $compression = unpack("Vcomp", substr($data, 0, 4));
02667     $data = substr($data, 4);
02668 
02669     //$compression = 0;
02670     if ($compression['comp'] != 0) {
02671       throw new Exception("$bitmap: compression not supported in bitmap image.\n");
02672     }
02673 
02674     // Remove bitmap data: data size, hres, vres, colours, imp. colours.
02675     $data = substr($data, 20);
02676 
02677     // Add the BITMAPCOREHEADER data
02678     $header  = pack("Vvvvv", 0x000c, $width, $height, 0x01, 0x18);
02679     $data = $header . $data;
02680 
02681     return (array($width, $height, $size, $data));
02682   }
