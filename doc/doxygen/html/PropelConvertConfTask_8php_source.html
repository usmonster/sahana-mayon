00161       if ( count($child) == 0 ) {

00163 
00164       }
00165 
00166       // add the childs attributes as if they where children
00167       foreach ( $v->attributes() as $ak => $av ) {
00168 
00169         // if the child is not an array, transform it into one
00170         if ( !is_array( $child ) ) {
00171           $child = array( "value" => $child );
00172         }
00173 
00174         if ($ak == 'id') {
00175           // special exception: if there is a key named 'id'
00176           // then we will name the current key after that id
00177           $k = self::getConvertedXmlValue($av);
00178         } else {
00179           // otherwise, just add the attribute like a child element
00180           $child[$ak] = self::getConvertedXmlValue($av);
00181         }
00182       }
00183 
00184       // if the $k is already in our children list, we need to transform
00185       // it into an array, else we add it as a value
00186       if ( !in_array( $k, array_keys($ar) ) ) {
00187         $ar[$k] = $child;
00188       } else {
00189         // (This only applies to nested nodes that do not have an @id attribute)
00190 
00191         // if the $ar[$k] element is not already an array, then we need to make it one.
00192         // this is a bit of a hack, but here we check to also make sure that if it is an
00193         // array, that it has numeric keys.  this distinguishes it from simply having other
00194         // nested element data.
00195 
00196         if ( !is_array($ar[$k]) || !isset($ar[$k][0]) ) { $ar[$k] = array($ar[$k]); }
