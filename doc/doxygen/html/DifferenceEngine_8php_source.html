00379         $i = 0;
00380         $j = 0;
00381 
00382         USE_ASSERTS && assert('count($lines) == count($changed)');
00383         $len = count($lines);
00384         $other_len = count($other_changed);
00385 
00386         while (1) {
00387             /*
00388              * Scan forwards to find beginning of another run of changes.
00389              * Also keep track of the corresponding point in the other file.
00390              *
00391              * Throughout this code, $i and $j are adjusted together so that
00392              * the first $i elements of $changed and the first $j elements
00393              * of $other_changed both contain the same number of zeros
00394              * (unchanged lines).
00395              * Furthermore, $j is always kept so that $j == $other_len or
00396              * $other_changed[$j] == false.
00397              */

00399                 $j++;
00400 
00401             while ($i < $len && ! $changed[$i]) {
00402                 USE_ASSERTS && assert('$j < $other_len && ! $other_changed[$j]');
00403                 $i++;
00404                 $j++;
00405                 while ($j < $other_len && $other_changed[$j])
00406                     $j++;
00407             }
00408 
00409             if ($i == $len)
00410                 break;
00411 
00412             $start = $i;
00413 
00414             // Find the end of this run of changes.
00415             while (++$i < $len && $changed[$i])
00416                 continue;
00417 
00418             do {
00419                 /*
00420                  * Record the length of this run of changes, so that
00421                  * we can later determine whether the run has grown.
00422                  */
00423                 $runlength = $i - $start;
00424 
00425                 /*
00426                  * Move the changed region back, so long as the
00427                  * previous unchanged line matches the last changed one.
00428                  * This merges with previous changed regions.
00429                  */
00430                 while ($start > 0 && $lines[$start - 1] == $lines[$i - 1]) {
00431                     $changed[--$start] = 1;
00432                     $changed[--$i] = false;
00433                     while ($start > 0 && $changed[$start - 1])
00434                         $start--;
00435                     USE_ASSERTS && assert('$j > 0');
00436                     while ($other_changed[--$j])
00437                         continue;
00438                     USE_ASSERTS && assert('$j >= 0 && !$other_changed[$j]');
00439                 }
00440 
00441                 /*
00442                  * Set CORRESPONDING to the end of the changed run, at the last
00443                  * point where it corresponds to a changed run in the other file.
00444                  * CORRESPONDING == LEN means no such point has been found.
00445                  */

00448                 /*
00449                  * Move the changed region forward, so long as the
00450                  * first changed line matches the following unchanged one.
00451                  * This merges with following changed regions.
