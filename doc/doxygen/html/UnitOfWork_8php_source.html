00662     public function buildFlushTree(array $tables)
00663     {
00664         // determine classes to order. only necessary because the $tables param
00665         // can contain strings or table objects...
00666         $classesToOrder = array();
00667         foreach ($tables as $table) {
00668             if ( ! ($table instanceof Doctrine_Table)) {
00669                 $table = $this->conn->getTable($table, false);
00670             }
00671             $classesToOrder[] = $table->getComponentName();
00672         }
00673         $classesToOrder = array_unique($classesToOrder);
00674 
00675         if (count($classesToOrder) < 2) {
00676             return $classesToOrder;
00677         }
00678 
00679         // build the correct order
00680         $flushList = array();
00681         foreach ($classesToOrder as $class) {
00682             $table = $this->conn->getTable($class, false);
00683             $currentClass = $table->getComponentName();
00684 
00685             $index = array_search($currentClass, $flushList);
00686 
00687             if ($index === false) {
00688                 //echo "adding $currentClass to flushlist";
00689                 $flushList[] = $currentClass;
00690                 $index = max(array_keys($flushList));
00691             }
00692 
00693             $rels = $table->getRelations();
00694 
00695             // move all foreignkey relations to the beginning
00696             foreach ($rels as $key => $rel) {
00697                 if ($rel instanceof Doctrine_Relation_ForeignKey) {
00698                     unset($rels[$key]);
00699                     array_unshift($rels, $rel);
00700                 }
00701             }
00702 
00703             foreach ($rels as $rel) {
00704                 $relatedClassName = $rel->getTable()->getComponentName();
00705 
00706                 if ( ! in_array($relatedClassName, $classesToOrder)) {
00707                     continue;
00708                 }
00709 
00710                 $relatedCompIndex = array_search($relatedClassName, $flushList);
00711                 $type = $rel->getType();
00712 
00713                 // skip self-referenced relations
00714                 if ($relatedClassName === $currentClass) {
00715                     continue;
00716                 }
00717 
00718                 if ($rel instanceof Doctrine_Relation_ForeignKey) {
00719                     // the related component needs to come after this component in
00720                     // the list (since it holds the fk)
00721 
00722                     if ($relatedCompIndex !== false) {
00723                         // the component is already in the list
00724                         if ($relatedCompIndex >= $index) {
00725                             // it's already in the right place
00726                             continue;
00727                         }
00728 
00729                         unset($flushList[$index]);
00730                         // the related comp has the fk. so put "this" comp immediately
00731                         // before it in the list
00732                         array_splice($flushList, $relatedCompIndex, 0, $currentClass);
00733                         $index = $relatedCompIndex;
00734                     } else {
00735                         $flushList[] = $relatedClassName;
00736                     }
00737 
00738                 } else if ($rel instanceof Doctrine_Relation_LocalKey) {
00739                     // the related component needs to come before the current component
00740                     // in the list (since this component holds the fk).
00741 
00742                     if ($relatedCompIndex !== false) {
00743                         // already in flush list
00744                         if ($relatedCompIndex <= $index) {
00745                             // it's in the right place
00746                             continue;
00747                         }
00748 
00749                         unset($flushList[$relatedCompIndex]);
00750                         // "this" comp has the fk. so put the related comp before it
00751                         // in the list
00752                         array_splice($flushList, $index, 0, $relatedClassName);
00753                     } else {
00754                         array_unshift($flushList, $relatedClassName);
00755                         $index++;
00756                     }
00757                 } else if ($rel instanceof Doctrine_Relation_Association) {
00758                     // the association class needs to come after both classes
00759                     // that are connected through it in the list (since it holds
00760                     // both fks)
00761 
00762                     $assocTable = $rel->getAssociationFactory();
00763                     $assocClassName = $assocTable->getComponentName();
00764 
00765                     if ($relatedCompIndex !== false) {
00766                         unset($flushList[$relatedCompIndex]);
00767                     }
00768 
00769                     array_splice($flushList, $index, 0, $relatedClassName);
00770                     $index++;
00771 
00772                     $index3 = array_search($assocClassName, $flushList);
00773 
00774                     if ($index3 !== false) {
00775                         if ($index3 >= $index) {
00776                             continue;
00777                         }
00778 
00779                         unset($flushList[$index3]);
00780                         array_splice($flushList, $index - 1, 0, $assocClassName);
00781                         $index = $relatedCompIndex;
00782                     } else {
00783                         $flushList[] = $assocClassName;
00784                     }
00785                 }
00786             }
00787         }
00788
