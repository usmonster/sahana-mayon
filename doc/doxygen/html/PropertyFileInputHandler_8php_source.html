00091     public void handleInput(InputRequest request) throws BuildException {
00092         readProps();
00093         
00094         Object o = props.get(request.getPrompt());
00095         if (o == null) {
00096             throw new BuildException("Unable to find input for \'"

00098         }
00099         request.setInput(o.toString());
00100         if (!request.isInputValid()) {
00101             throw new BuildException("Found invalid input " + o
00102                                      + " for \'" + request.getPrompt() + "\'");
00103         }
00104     }

00118
