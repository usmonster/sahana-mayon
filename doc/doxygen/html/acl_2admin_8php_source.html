00282     function _tree_sort($a,$b){
00283         // handle the trivial cases first
00284         if ($a['id'] == '') return -1;
00285         if ($b['id'] == '') return 1;
00286         // split up the id into parts
00287         $a_ids = explode(':', $a['id']);
00288         $b_ids = explode(':', $b['id']);
00289         // now loop through the parts
00290         while (count($a_ids) && count($b_ids)) {
00291             // compare each level from upper to lower
00292             // until a non-equal component is found
00293             $cur_result = strcmp(array_shift($a_ids), array_shift($b_ids));
00294             if ($cur_result) {
00295                 // if one of the components is the last component and is a file
00296                 // and the other one is either of a deeper level or a directory,
00297                 // the file has to come after the deeper level or directory
00298                 if (empty($a_ids) && $a['type'] == 'f' && (count($b_ids) || $b['type'] == 'd')) return 1;
00299                 if (empty($b_ids) && $b['type'] == 'f' && (count($a_ids) || $a['type'] == 'd')) return -1;
00300                 return $cur_result;
00301             }
00302         }
00303         // The two ids seem to be equal. One of them might however refer
00304         // to a page, one to a namespace, the namespace needs to be first.
00305         if (empty($a_ids) && empty($b_ids)) {
00306             if ($a['type'] == $b['type']) return 0;
00307             if ($a['type'] == 'f') return 1;
00308             return -1;
00309         }
00310         // Now the empty part is either a page in the parent namespace
00311         // that obviously needs to be after the namespace
00312         // Or it is the namespace that contains the other part and should be
00313         // before that other part.
00314         if (empty($a_ids)) return ($a['type'] == 'd') ? -1 : 1;
00315         if (empty($b_ids)) return ($b['type'] == 'd') ? 1 : -1;
00316     }
