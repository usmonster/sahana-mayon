01244   private function _calculateSpans()

01247     // In HTML only the upper-left cell should be written and it should have

01254 
01255       $candidateSpannedRow  = array();
01256 
01257       // loop through all Excel merged cells
01258       foreach ($sheet->getMergeCells() as $cells) {
01259         list($cells, ) = PHPExcel_Cell::splitRange($cells);
01260         $first = $cells[0];
01261         $last  = $cells[1];
01262 
01263         list($fc, $fr) = PHPExcel_Cell::coordinateFromString($first);
01264         $fc = PHPExcel_Cell::columnIndexFromString($fc) - 1;
01265 
01266         list($lc, $lr) = PHPExcel_Cell::coordinateFromString($last);
01267         $lc = PHPExcel_Cell::columnIndexFromString($lc) - 1;
01268 
01269         // loop through the individual cells in the individual merge
01270         for ($r = $fr; $r <= $lr; ++$r) {
01271           // also, flag this row as a HTML row that is candidate to be omitted
01272           $candidateSpannedRow[$r] = $r;
01273 
01274           for ($c = $fc; $c <= $lc; ++$c) {
01275             if ( !($c == $fc && $r == $fr) ) {
01276               // not the upper-left cell (should not be written in HTML)
01277               $this->_isSpannedCell[$sheetIndex][$r][$c] = array(
01278                 'baseCell' => array($fr, $fc),
01279               );
01280             } else {
01281               // upper-left is the base cell that should hold the colspan/rowspan attribute
01282               $this->_isBaseCell[$sheetIndex][$r][$c] = array(
01283                 'xlrowspan' => $lr - $fr + 1, // Excel rowspan
01284                 'rowspan'   => $lr - $fr + 1, // HTML rowspan, value may change
01285                 'xlcolspan' => $lc - $fc + 1, // Excel colspan
01286                 'colspan'   => $lc - $fc + 1, // HTML colspan, value may change
01287               );
01288             }
01289           }
01290         }
01291       }
01292 
01293       // Identify which rows should be omitted in HTML. These are the rows where all the cells
01294       //   participate in a merge and the where base cells are somewhere above.

01299             $this->_isSpannedRow[$sheetIndex][$rowIndex] = $rowIndex;
01300           };
01301         }
01302       }
01303 
01304       // For each of the omitted rows we found above, the affected rowspans should be subtracted by 1
01305       if ( isset($this->_isSpannedRow[$sheetIndex]) ) {
