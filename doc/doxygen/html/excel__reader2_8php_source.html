00942   function _format_value($format, $num, $f)
00943   {
00944     // 49==TEXT format
00945     // http://code.google.com/p/php-excel-reader/issues/detail?id=7
00946     if ((!$f && $format == "%s") || ($f == 49) || ($format == "GENERAL")) {
00947       return array('string' => $num, 'formatColor' => null);
00948     }
00949 
00950     // Custom pattern can be POSITIVE;NEGATIVE;ZERO
00951     // The "text" option as 4th parameter is not handled
00952     $parts = split(";", $format);
00953     $pattern = $parts[0];
00954     // Negative pattern
00955     if (count($parts) > 2 && $num == 0) {
00956       $pattern = $parts[2];
00957     }
00958     // Zero pattern
00959     if (count($parts) > 1 && $num < 0) {
00960       $pattern = $parts[1];
00961       $num = abs($num);
00962     }
00963 
00964     $color = "";
00965     $matches = array();
00966     $color_regex = "/^\[(BLACK|BLUE|CYAN|GREEN|MAGENTA|RED|WHITE|YELLOW)\]/i";
00967     if (preg_match($color_regex, $pattern, $matches)) {
00968       $color = strtolower($matches[1]);
00969       $pattern = preg_replace($color_regex, "", $pattern);
00970     }
00971 
00972     // In Excel formats, "_" is used to add spacing, which we can't do in HTML
00973     $pattern = preg_replace("/_./", "", $pattern);
00974 
00975     // Some non-number characters are escaped with \, which we don't need
00976     $pattern = preg_replace("/\\\/", "", $pattern);
00977 
00978     // Some non-number strings are quoted, so we'll get rid of the quotes
00979     $pattern = preg_replace("/\"/", "", $pattern);
00980 
00981     // TEMPORARY - Convert # to 0
00982     $pattern = preg_replace("/\#/", "0", $pattern);
00983 
00984     // Find out if we need comma formatting
00985     $has_commas = preg_match("/,/", $pattern);
00986     if ($has_commas) {
00987       $pattern = preg_replace("/,/", "", $pattern);
00988     }
00989 
00990     // Handle Percentages
00991     if (preg_match("/\d(\%)([^\%]|$)/", $pattern, $matches)) {
00992       $num = $num * 100;
00993       $pattern = preg_replace("/(\d)(\%)([^\%]|$)/", "$1%$3", $pattern);
00994     }
00995 
00996     // Handle the number itself
00997     $number_regex = "/(\d+)(\.?)(\d*)/";
00998     if (preg_match($number_regex, $pattern, $matches)) {
00999       $left = $matches[1];
01000       $dec = $matches[2];

01133 
01134     $code = v($data, $pos);
01135     $length = v($data, $pos + 2);
01136 
01137     while ($code != SPREADSHEET_EXCEL_READER_TYPE_EOF) {
01138       switch ($code) {
01139         case SPREADSHEET_EXCEL_READER_TYPE_SST:
01140           $spos = $pos + 4;
01141           $limitpos = $spos + $length;
01142           $uniqueStrings = $this->_GetInt4d($data, $spos + 4);
01143           $spos += 8;
01144           for ($i = 0; $i < $uniqueStrings; $i++) {
01145             // Read in the number of characters
01146             if ($spos == $limitpos) {
01147               $opcode = v($data, $spos);
01148               $conlength = v($data, $spos + 2);
01149               if ($opcode != 0x3c) {
01150                 return -1;
01151               }
01152               $spos += 4;
01153               $limitpos = $spos + $conlength;
01154             }
01155             $numChars = ord($data[$spos]) | (ord($data[$spos + 1]) << 8);
01156             $spos += 2;
01157             $optionFlags = ord($data[$spos]);
01158             $spos++;
01159             $asciiEncoding = (($optionFlags & 0x01) == 0);
01160             $extendedString = ( ($optionFlags & 0x04) != 0);
01161 
01162             // See if string contains formatting information
01163             $richString = ( ($optionFlags & 0x08) != 0);
01164 
01165             if ($richString) {
01166               // Read in the crun
01167               $formattingRuns = v($data, $spos);
01168               $spos += 2;
01169             }
01170 
01171             if ($extendedString) {
01172               // Read in cchExtRst
01173               $extendedRunLength = $this->_GetInt4d($data, $spos);
01174               $spos += 4;
01175             }
01176 
01177             $len = ($asciiEncoding) ? $numChars : $numChars * 2;
01178             if ($spos + $len < $limitpos) {
01179               $retstr = substr($data, $spos, $len);
01180               $spos += $len;
01181             } else {
01182               // found countinue
01183               $retstr = substr($data, $spos, $limitpos - $spos);
01184               $bytesRead = $limitpos - $spos;
01185               $charsLeft = $numChars - (($asciiEncoding) ? $bytesRead : ($bytesRead / 2));
01186               $spos = $limitpos;
01187 
01188               while ($charsLeft > 0) {
01189                 $opcode = v($data, $spos);
01190                 $conlength = v($data, $spos + 2);
01191                 if ($opcode != 0x3c) {
01192                   return -1;
01193                 }
01194                 $spos += 4;
01195                 $limitpos = $spos + $conlength;
01196                 $option = ord($data[$spos]);
01197                 $spos += 1;
01198                 if ($asciiEncoding && ($option == 0)) {
01199                   $len = min($charsLeft, $limitpos - $spos); // min($charsLeft, $conlength);
01200                   $retstr .= substr($data, $spos, $len);
01201                   $charsLeft -= $len;
01202                   $asciiEncoding = true;
01203                 } elseif (!$asciiEncoding && ($option != 0)) {
01204                   $len = min($charsLeft * 2, $limitpos - $spos); // min($charsLeft, $conlength);
01205                   $retstr .= substr($data, $spos, $len);
01206                   $charsLeft -= $len / 2;
01207                   $asciiEncoding = false;
01208                 } elseif (!$asciiEncoding && ($option == 0)) {
01209                   // Bummer - the string starts off as Unicode, but after the
01210                   // continuation it is in straightforward ASCII encoding
01211                   $len = min($charsLeft, $limitpos - $spos); // min($charsLeft, $conlength);
01212                   for ($j = 0; $j < $len; $j++) {
01213                     $retstr .= $data[$spos + $j] . chr(0);
01214                   }
01215                   $charsLeft -= $len;
01216                   $asciiEncoding = false;
01217                 } else {
01218                   $newstr = '';
01219                   for ($j = 0; $j < strlen($retstr); $j++) {
01220                     $newstr = $retstr[$j] . chr(0);
01221                   }
01222                   $retstr = $newstr;
01223                   $len = min($charsLeft * 2, $limitpos - $spos); // min($charsLeft, $conlength);
01224                   $retstr .= substr($data, $spos, $len);
01225                   $charsLeft -= $len / 2;
01226                   $asciiEncoding = false;
01227                 }
01228                 $spos += $len;
01229               }
01230             }

01232

01235             }
01236

01239               $spos += $extendedRunLength;
