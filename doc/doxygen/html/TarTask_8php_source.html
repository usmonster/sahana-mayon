00315         
00316         // we'll need to add to that list of implicit dirs any directories

00320           foreach($incDirs as $dircheck) {
00321             if (!empty($dir) && $dir == dirname($dircheck)) {

00324           }
00325         }
00326         
00327         $implicitDirs = array_unique($implicitDirs);
00328         
00329         // Now add any empty dirs (dirs not covered by the implicit dirs)
00330         // to the files array.

00332         foreach($incDirs as $dir) { // we cannot simply use array_diff() since we want to disregard empty/. dirs
00333           if ($dir != "" && $dir != "." && !in_array($dir, $implicitDirs)) {
00334             // it's an empty dir, so we'll add it.
00335             $this->files[] = $dir;
00336           }
