00072   public function save($file = null, $fileMode = 0666, $create = true, $dirMode = 0777)
00073   {
00074     if (null === $file)
00075     {
00076       $file = $this->generateFilename();
00077     }
00078 
00079     if ($file[0] != '/' && $file[0] != '\\' && !(strlen($file) > 3 && ctype_alpha($file[0]) && $file[1] == ':' && ($file[2] == '\\' || $file[2] == '/')))
00080     {
00081       if (null === $this->path)
00082       {
00083         throw new RuntimeException('You must give a "path" when you give a relative file name.');
00084       }
00085 
00086       $file = $this->path.DIRECTORY_SEPARATOR.$file;
00087     }
00088 
00089     // get our directory path from the destination filename
00090     $directory = dirname($file);
00091 
00092     if (!is_readable($directory))
00093     {
00094       if ($create && !@mkdir($directory, $dirMode, true))
00095       {
00096         // failed to create the directory
00097         throw new Exception(sprintf('Failed to create file upload directory "%s".', $directory));
00098       }
00099 
00100       // chmod the directory since it doesn't seem to work on recursive paths
00101       chmod($directory, $dirMode);
00102     }
00103 
00104     if (!is_dir($directory))
00105     {
00106       // the directory path exists but it's not a directory
00107       throw new Exception(sprintf('File upload path "%s" exists, but is not a directory.', $directory));
00108     }
00109 
00110     if (!is_writable($directory))
00111     {
00112       // the directory isn't writable
00113       throw new Exception(sprintf('File upload path "%s" is not writable.', $directory));
00114     }
00115 
00116     // copy the temp file to the destination file
00117     copy($this->getTempName(), $file);
00118 
00119     // chmod our file
00120     chmod($file, $fileMode);
00121 
00122     $this->savedName = $file;
00123 
00124     return null === $this->path ? $file : str_replace($this->path.DIRECTORY_SEPARATOR, '', $file);
00125   }
