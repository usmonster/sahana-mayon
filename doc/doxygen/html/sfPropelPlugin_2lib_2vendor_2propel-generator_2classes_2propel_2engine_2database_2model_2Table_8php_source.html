00346   public function doFinalInitialization()
00347   {
00348     // Heavy indexing must wait until after all columns composing
00349     // a table's primary key have been parsed.
00350     if ($this->heavyIndexing) {
00351       $this->doHeavyIndexing();
00352     }
00353 
00354     // Name any indices which are missing a name using the
00355     // appropriate algorithm.
00356     $this->doNaming();
00357 
00358     // execute behavior table modifiers
00359     foreach ($this->getBehaviors() as $behavior)
00360     {
00361       $behavior->getTableModifier()->modifyTable();
00362     }
00363     
00364     // if idMethod is "native" and in fact there are no autoIncrement
00365     // columns in the table, then change it to "none"
00366     $anyAutoInc = false;
00367     foreach ($this->getColumns() as $col) {
00368       if ($col->isAutoIncrement()) {
00369         $anyAutoInc = true;
00370       }
00371     }
00372     if ($this->getIdMethod() === IDMethod::NATIVE && !$anyAutoInc) {
00373       $this->setIdMethod(IDMethod::NO_ID_METHOD);
00374     }
00375     
00376     // If there is no PK, then throw an error. Propel 1.3 requires primary keys.
00377     $pk = $this->getPrimaryKey();
00378     if (empty($pk)) {
00379       throw new EngineException("Table '".$this->getName()."' does not have a primary key defined.  Propel requires all tables to have a primary key.");
00380     }

00401   private function doHeavyIndexing()
00402   {
00403     if (self::DEBUG) {
00404       print("doHeavyIndex() called on table " . $this->name."\n");
00405     }
00406 
00407     $pk = $this->getPrimaryKey();
00408     $size = count($pk);
00409 
00410     // We start at an offset of 1 because the entire column
00411     // list is generally implicitly indexed by the fact that
00412     // it's a primary key.
00413     for ($i=1; $i < $size; $i++) {
00414       $idx = new Index();
00415       $idx->setColumns(array_slice($pk, $i, $size));
00416       $this->addIndex($idx);
00417     }

00424   private function doNaming() {
00425 
00426     // Assure names are unique across all databases.
00427     try {
00428       for ($i=0, $size = count($this->foreignKeys); $i < $size; $i++) {
00429         $fk = $this->foreignKeys[$i];
00430         $name = $fk->getName();
00431         if (empty($name)) {
00432           $name = $this->acquireConstraintName("FK", $i + 1);
00433           $fk->setName($name);
00434         }
00435       }
00436 
00437       for ($i = 0, $size = count($this->indices); $i < $size; $i++) {
00438         $index = $this->indices[$i];
00439         $name = $index->getName();
00440         if (empty($name)) {
00441           $name = $this->acquireConstraintName("I", $i + 1);
00442           $index->setName($name);
00443         }
00444       }
00445 
00446       for ($i = 0, $size = count($this->unices); $i < $size; $i++) {
00447         $index = $this->unices[$i];
00448         $name = $index->getName();
00449         if (empty($name)) {
00450           $name = $this->acquireConstraintName("U", $i + 1);
00451           $index->setName($name);
00452         }
00453       }
00454 
00455       // NOTE: Most RDBMSes can apparently name unique column
00456       // constraints/indices themselves (using MySQL and Oracle
00457       // as test cases), so we'll assume that we needn't add an
00458       // entry to the system name list for these.
00459     } catch (EngineException $nameAlreadyInUse) {
00460       print $nameAlreadyInUse->getMessage() . "\n";
00461       print $nameAlreadyInUse->getTraceAsString();
00462     }
00463   }
