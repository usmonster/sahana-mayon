00076     
00077         if ( !$this->getInitialized() ) {
00078             $this->_initialize();

00083

00087         
00088         $lines = explode("\n", $buffer);        
00089         $filtered = array();    
00090             
00091         $commentsSize = count($this->_comments);
00092         
00093         foreach($lines as $line) {            
00094             for($i = 0; $i < $commentsSize; $i++) {
00095                 $comment = $this->_comments[$i]->getValue();
00096                 if ( StringHelper::startsWith($comment, ltrim($line)) ) {
00097                     $line = null;
00098                     break;
00099                 }
00100             }
00101             if ($line !== null) {
00102                 $filtered[] = $line;
00103             }
00104         }
00105                 
00106         $filtered_buffer = implode("\n", $filtered);    
00107         return $filtered_buffer;
