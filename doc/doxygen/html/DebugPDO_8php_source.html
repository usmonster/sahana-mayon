00384     if (!$this->getLoggingConfig('enabled', true))

00389       return;

00398             $now = $this->getDebugSnapshot();

00400         }

00402     // If the necessary additional parameters were given, get the debug log prefix for the log line
00403     if ($methodName && $debugSnapshot)
00404       $msg = $this->getLogPrefix($methodName, $debugSnapshot) . $msg;

00406     // We won't log empty messages
